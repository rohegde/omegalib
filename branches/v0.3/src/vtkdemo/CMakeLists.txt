#######################################################################################################################
FIND_PACKAGE(VTK)
IF(NOT VTK_DIR)
  MESSAGE(FATAL_ERROR "Please set VTK_DIR.")
ENDIF(NOT VTK_DIR)
INCLUDE(${VTK_USE_FILE})

#######################################################################################################################
# Compile definitions
ADD_DEFINITIONS( )

#######################################################################################################################
# Source files
SET( srcs 
        vtkdemo.cpp
		VtkDemoApplication.cpp
        )
    
#######################################################################################################################
# Headers
SET( headers 
		VtkDemoApplication.h
        ) 
        
#######################################################################################################################
# Setup compile info
ADD_EXECUTABLE( vtkdemo ${srcs} ${headers})

#######################################################################################################################
# Setup link info.
TARGET_LINK_LIBRARIES( vtkdemo
	omega 
	outils
    vtkRendering
    vtkGraphics
    vtkIO
    vtkCommon
    vtkHybrid
	)
